async function generateCV() { const { jsPDF } = window.jspdf; const name = document.getElementById('cv_name').value || "بدون اسم"; const phone = document.getElementById('cv_phone').value || "-"; const email = document.getElementById('cv_email').value || "-"; const summary = document.getElementById('cv_summary').value || "-"; const details = document.getElementById('cv_details').value || "-"; const lang = document.getElementById('cv_lang').value; const tempDiv = document.createElement('div'); tempDiv.style.width = '700px'; tempDiv.style.padding = '40px'; tempDiv.style.background = 'white'; tempDiv.style.color = 'black'; tempDiv.style.direction = lang === 'ar' ? 'rtl' : 'ltr'; tempDiv.style.fontFamily = "Tajawal"; tempDiv.innerHTML = `

${name}

${phone} | ${email}

${lang === 'ar' ? 'النبذة التعريفية' : 'Professional Summary'}

${summary}

${lang === 'ar' ? 'المؤهلات والخبرات' : 'Experience & Education'}

${details}

`; document.body.appendChild(tempDiv); const canvas = await html2canvas(tempDiv, { scale: 2 }); const imgData = canvas.toDataURL('image/png'); const pdf = new jsPDF(); pdf.addImage(imgData, 'PNG', 0, 0, 210, 297); pdf.save('CV_Professional.pdf'); document.body.removeChild(tempDiv); }
